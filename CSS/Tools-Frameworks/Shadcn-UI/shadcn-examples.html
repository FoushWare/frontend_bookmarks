Form Component

Form with validation using react-hook-form

import { useForm } from "react-hook-form" import { Form, FormControl, FormField, FormItem, FormLabel } from "@/components/ui/form" import { Input } from "@/components/ui/input" const form = useForm({ defaultValues: { username: "", email: "", }, }) <Form {...form}> <FormField control={form.control} name="username" render={({ field }) => ( <FormItem> <FormLabel>Username</FormLabel> <FormControl> <Input placeholder="Username" {...field} /> </FormControl> </FormItem> )} /> </Form>
